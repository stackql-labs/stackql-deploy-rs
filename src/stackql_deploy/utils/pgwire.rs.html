3//! Pure-Rust PostgreSQL simple-query wire protocol client.
4//!
5//! Implements only what stackql-deploy needs: unencrypted TCP connections
6//! to a local StackQL server using the PostgreSQL simple query protocol (v3).
7//! No native dependencies (replaces pgwire-lite → libpq-sys).
8
9use std::collections::{HashMap, HashSet};
10use std::io::{Read, Write};
11use std::net::TcpStream;

13/// A single column value returned from a query.
14pub enum Value {
15    String(String),
16    Null,
17    Bool(bool),
18    Integer(i64),
19    Float(f64),
20    Bytes(Vec<u8>),
21}

23/// A server notice (NOTICE, WARNING, etc.).
24pub struct Notice {
25    pub fields: HashMap<String, String>,
26}

28/// The result of a [`PgwireLite::query`] call.
29pub struct PgQueryResult {
30    pub column_names: Vec<String>,
31    pub rows: Vec<HashMap<String, Value>>,
32    pub notices: Vec<Notice>,
33    /// Row count reported by CommandComplete (INSERT/UPDATE/DELETE n).
34    pub row_count: usize,
35}

37/// Minimal PostgreSQL wire-protocol client.
38pub struct PgwireLite {
39    stream: TcpStream,
40    /// Canonical signatures of every notice line surfaced earlier in this
41    /// session. stackql emits each new query's NoticeResponse with a
42    /// cumulative `detail` field containing every provider notice seen so
43    /// far. Any detail line already present in this set is stale and
44    /// dropped from subsequent query results.
45    seen_notice_sigs: HashSet<String>,
46}
47
48impl PgwireLite {
49    /// Connect to a PostgreSQL-protocol server (e.g. StackQL) at `host:port`.
50    ///
51    /// `_ssl` and `_verbosity` are accepted for API compatibility but ignored;
52    /// the connection is always unencrypted (StackQL default).
53    pub fn new(host: &str, port: u16, _ssl: bool, _verbosity: &str) -> Result<Self, String> {
54        let addr = format!("{}:{}", host, port);
55        let stream = TcpStream::connect(&addr)
56            .map_err(|e| format!("Connection to {} failed: {}", addr, e))?;
57
58        let mut client = PgwireLite {
59            stream,
60            seen_notice_sigs: HashSet::new(),
61        };
62        client.startup()?;
63        Ok(client)
64    }
65
66    /// Returns a version string (no libpq; just identifies the client).
67    pub fn libpq_version(&self) -> String {
68        "pure-rust-pgwire-client".to_string()
69    }

71    // ------------------------------------------------------------------
72    // Startup handshake
73    // ------------------------------------------------------------------
74
75    fn startup(&mut self) -> Result<(), String> {
76        // Protocol version 3.0 = 0x00_03_00_00
77        const PROTOCOL_V3: i32 = 196608;
78
79        // Startup message: user=stackql, database=stackql, then double-null
80        let params = b"user\0stackql\0database\0stackql\0\0";
81        let total_len = 4 + 4 + params.len(); // length field + protocol + params
82
83        let mut msg = Vec::with_capacity(total_len);
84        msg.extend_from_slice(&(total_len as i32).to_be_bytes());
85        msg.extend_from_slice(&PROTOCOL_V3.to_be_bytes());
86        msg.extend_from_slice(params);
87
88        self.stream
89            .write_all(&msg)
90            .map_err(|e| format!("Startup write error: {}", e))?;
91
92        // Process auth / parameter-status messages until ReadyForQuery
93        loop {
94            let msg_type = self.read_byte()?;
95            let payload_len = self.read_i32()? as usize;
96            // payload_len includes the 4 bytes of the length field itself
97            let data = self.read_bytes(payload_len.saturating_sub(4))?;
98
99            match msg_type {
100                b'R' => {
101                    // AuthenticationRequest
102                    let auth_type =
103                        i32::from_be_bytes(data[..4].try_into().map_err(|_| "Bad auth")?);
104                    if auth_type != 0 {
105                        return Err(format!(
106                            "Unsupported authentication type {} from server",
107                            auth_type
108                        ));
109                    }
110                    // AuthenticationOk — nothing to do
111                }
112                b'K' => {}     // BackendKeyData — ignore
113                b'S' => {}     // ParameterStatus — ignore
114                b'Z' => break, // ReadyForQuery
115                b'E' => return Err(parse_error_fields(&data)),
116                b'N' => {} // NoticeResponse during startup — ignore
117                _ => {}    // Unknown message type — skip
118            }
119        }
120
121        Ok(())
122    }

128    /// Execute a simple (non-prepared) SQL query and return structured results.
129    pub fn query(&mut self, sql: &str) -> Result<PgQueryResult, String> {
130        // Drain any bytes the server may have emitted outside a prior query's
131        // response window. stackql has been observed to re-emit stale
132        // NoticeResponse frames from earlier statements, which would
133        // otherwise be attributed to this query.
134        self.drain_pending();
135
136        // Send Query message: 'Q' | int32(len) | sql\0
137        let sql_bytes = sql.as_bytes();
138        let payload_len = 4 + sql_bytes.len() + 1; // length field + sql + null
139
140        let mut msg = Vec::with_capacity(1 + payload_len);
141        msg.push(b'Q');
142        msg.extend_from_slice(&(payload_len as i32).to_be_bytes());
143        msg.extend_from_slice(sql_bytes);
144        msg.push(0u8);
145
146        self.stream
147            .write_all(&msg)
148            .map_err(|e| format!("Query write error: {}", e))?;
149
150        // Collect response messages
151        let mut column_names: Vec<String> = Vec::new();
152        let mut rows: Vec<HashMap<String, Value>> = Vec::new();
153        let mut notices: Vec<Notice> = Vec::new();
154        let mut row_count: usize = 0;
155
156        loop {
157            let msg_type = self.read_byte()?;
158            let payload_len = self.read_i32()? as usize;
159            let data = self.read_bytes(payload_len.saturating_sub(4))?;
160
161            match msg_type {
162                b'T' => {
163                    // RowDescription
164                    column_names = parse_row_description(&data);
165                }
166                b'D' => {
167                    // DataRow
168                    let row = parse_data_row(&data, &column_names);
169                    rows.push(row);
170                }
171                b'C' => {
172                    // CommandComplete — tag like "SELECT 5", "INSERT 0 1", "UPDATE 3"
173                    let tag = std::str::from_utf8(data.strip_suffix(b"\0").unwrap_or(&data))
174                        .unwrap_or("")
175                        .to_string();
176                    if let Some(n) = tag.split_whitespace().last().and_then(|s| s.parse().ok()) {
177                        row_count = n;
178                    }
179                }
180                b'N' => {
181                    notices.push(parse_notice_fields(&data));
182                }
183                b'E' => {
184                    // Capture the error but DON'T return yet — we must
185                    // drain the stream until ReadyForQuery ('Z') so the
186                    // connection is left in a clean state for the next query.
187                    let err_msg = parse_error_fields(&data);
188                    // Continue reading until ReadyForQuery
189                    loop {
190                        let drain_type = self.read_byte()?;
191                        let drain_len = self.read_i32()? as usize;
192                        let _drain_data = self.read_bytes(drain_len.saturating_sub(4))?;
193                        if drain_type == b'Z' {
194                            break;
195                        }
196                    }
197                    return Err(err_msg);
198                }
199                b'I' => {}     // EmptyQueryResponse
200                b'Z' => break, // ReadyForQuery — done
201                _ => {}
202            }
203        }
204
205        // Drop any detail lines we've already surfaced earlier in this
206        // session; stackql keeps re-emitting them on every subsequent
207        // query's NoticeResponse.
208        let kept = filter_stale_notices(notices, &mut self.seen_notice_sigs);
209
210        Ok(PgQueryResult {
211            column_names,
212            rows,
213            notices: kept,
214            row_count,
215        })
216    }
217
218    /// Discard any bytes the server sent outside a query response window.
219    ///
220    /// The server is expected to stay silent between queries (the prior
221    /// response ended with `ReadyForQuery` / `'Z'`). stackql has been
222    /// observed to emit stray `NoticeResponse` frames referring to
223    /// previously-executed statements; if left in the socket buffer those
224    /// frames would be read at the start of the next query and
225    /// misattributed. Best-effort non-blocking read; any I/O error is
226    /// swallowed since the normal path is zero bytes available.
227    fn drain_pending(&mut self) {
228        if self.stream.set_nonblocking(true).is_err() {
229            return;
230        }
231        let mut buf = [0u8; 4096];
232        loop {
233            match self.stream.read(&mut buf) {
234                Ok(0) => break,    // EOF
235                Ok(_) => continue, // keep draining
236                Err(_) => break,   // WouldBlock or other — nothing more pending
237            }
238        }
239        let _ = self.stream.set_nonblocking(false);
240    }

243    // Low-level I/O helpers
244    // ------------------------------------------------------------------
245
246    fn read_byte(&mut self) -> Result<u8, String> {
247        let mut buf = [0u8; 1];
248        self.stream
249            .read_exact(&mut buf)
250            .map_err(|e| format!("Read error: {}", e))?;
251        Ok(buf[0])
252    }
253
254    fn read_i32(&mut self) -> Result<i32, String> {
255        let mut buf = [0u8; 4];
256        self.stream
257            .read_exact(&mut buf)
258            .map_err(|e| format!("Read error: {}", e))?;
259        Ok(i32::from_be_bytes(buf))
260    }
261
262    fn read_bytes(&mut self, n: usize) -> Result<Vec<u8>, String> {
263        let mut buf = vec![0u8; n];
264        self.stream
265            .read_exact(&mut buf)
266            .map_err(|e| format!("Read error: {}", e))?;
267        Ok(buf)
268    }

271// ------------------------------------------------------------------
272// Stale notice filtering
273// ------------------------------------------------------------------
274
275/// Build a signature for a notice line. Dedup uses exact byte-match (after
276/// trimming) so two genuine provider errors — which always differ in their
277/// embedded request/serving IDs — are never conflated. A re-emitted stale
278/// notice is byte-identical to its first emission and dedups cleanly.
279fn canonical_line(line: &str) -> String {
280    line.trim().to_string()
281}
282
283/// Drop detail lines whose canonical signature was already surfaced earlier
284/// in the session. `seen` is mutated in place: every line that survives
285/// filtering is added so it counts as stale on subsequent queries. A notice
286/// whose message is stale and whose detail becomes empty after filtering is
287/// dropped entirely.
288fn filter_stale_notices(notices: Vec<Notice>, seen: &mut HashSet<String>) -> Vec<Notice> {
289    let mut kept = Vec::with_capacity(notices.len());
290
291    for n in notices {
292        let message_stale = n
293            .fields
294            .get("message")
295            .map(|m| seen.contains(&format!("M|{}", canonical_line(m))))
296            .unwrap_or(false);
297
298        let filtered_detail = n.fields.get("detail").map(|detail| {
299            detail
300                .lines()
301                .filter(|line| {
302                    let c = canonical_line(line);
303                    c.is_empty() || !seen.contains(&format!("D|{}", c))
304                })
305                .collect::<Vec<_>>()
306                .join("\n")
307        });
308
309        let has_detail_content = filtered_detail
310            .as_deref()
311            .map(|d| !d.trim().is_empty())
312            .unwrap_or(false);
313
314        if message_stale && !has_detail_content {
315            continue;
316        }
317
318        // Mark the surviving signatures as seen so they're filtered from
319        // future queries.
320        if let Some(m) = n.fields.get("message") {
321            let c = canonical_line(m);
322            if !c.is_empty() {
323                seen.insert(format!("M|{}", c));
324            }
325        }
326        if let Some(d) = filtered_detail.as_deref() {
327            for line in d.lines() {
328                let c = canonical_line(line);
329                if !c.is_empty() {
330                    seen.insert(format!("D|{}", c));
331                }
332            }
333        }
334
335        let mut new_fields = n.fields.clone();
336        if let Some(d) = filtered_detail {
337            if d.trim().is_empty() {
338                new_fields.remove("detail");
339            } else {
340                new_fields.insert("detail".to_string(), d);
341            }
342        }
343        kept.push(Notice { fields: new_fields });
344    }
345
346    kept
347}

353fn parse_row_description(data: &[u8]) -> Vec<String> {
354    let mut names = Vec::new();
355    if data.len() < 2 {
356        return names;
357    }
358    let num_fields = u16::from_be_bytes([data[0], data[1]]) as usize;
359    let mut pos = 2;
360
361    for _ in 0..num_fields {
362        // Null-terminated field name
363        let Some(null_off) = data[pos..].iter().position(|&b| b == 0) else {
364            break;
365        };
366        let name = String::from_utf8_lossy(&data[pos..pos + null_off]).into_owned();
367        names.push(name);
368        // Skip: name + null(1) + tableOID(4) + attrNum(2) + typeOID(4) + typeSize(2)
369        //       + typeMod(4) + formatCode(2) = 19 bytes after the null
370        pos += null_off + 1 + 18;
371    }
372    names
373}

375fn parse_data_row(data: &[u8], columns: &[String]) -> HashMap<String, Value> {
376    let mut row = HashMap::new();
377    if data.len() < 2 {
378        return row;
379    }
380    let num_cols = u16::from_be_bytes([data[0], data[1]]) as usize;
381    let mut pos = 2;
382
383    for col_name in columns.iter().take(num_cols) {
384        if pos + 4 > data.len() {
385            break;
386        }
387        let col_len = i32::from_be_bytes([data[pos], data[pos + 1], data[pos + 2], data[pos + 3]]);
388        pos += 4;
389
390        let value = if col_len < 0 {
391            Value::Null
392        } else {
393            let len = col_len as usize;
394            if pos + len > data.len() {
395                break;
396            }
397            let s = String::from_utf8_lossy(&data[pos..pos + len]).into_owned();
398            pos += len;
399            Value::String(s)
400        };
401
402        row.insert(col_name.clone(), value);
403    }
404    row
405}

407fn parse_notice_fields(data: &[u8]) -> Notice {
408    let mut fields = HashMap::new();
409    let mut pos = 0;
410
411    while pos < data.len() {
412        let field_code = data[pos];
413        pos += 1;
414        if field_code == 0 {
415            break;
416        }
417        let Some(null_off) = data[pos..].iter().position(|&b| b == 0) else {
418            break;
419        };
420        let value = String::from_utf8_lossy(&data[pos..pos + null_off]).into_owned();
421        pos += null_off + 1;
422
423        let key = match field_code {
424            b'S' => "severity",
425            b'M' => "message",
426            b'D' => "detail",
427            b'H' => "hint",
428            b'C' => "code",
429            b'P' => "position",
430            b'W' => "where",
431            _ => continue,
432        };
433        fields.insert(key.to_string(), value);
434    }
435
436    Notice { fields }
437}

439fn parse_error_fields(data: &[u8]) -> String {
440    let mut pos = 0;
441    while pos < data.len() {
442        let field_code = data[pos];
443        pos += 1;
444        if field_code == 0 {
445            break;
446        }
447        let Some(null_off) = data[pos..].iter().position(|&b| b == 0) else {
448            break;
449        };
450        let value = String::from_utf8_lossy(&data[pos..pos + null_off]).into_owned();
451        pos += null_off + 1;
452        if field_code == b'M' {
453            return value;
454        }
455    }
456    "Unknown server error".to_string()
